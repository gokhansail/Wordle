private async Task<List<Node>> LoadNodesFromTxtAsync(string filePath)
{
    var nodeList = new List<Node>();
    using StreamReader reader = new StreamReader(filePath);
    string line;
    while ((line = await reader.ReadLineAsync()) != null)
    {
        string[] parts = line.Split(' ');
        if (parts.Length != 4) continue;
        nodeList.Add(new Node
        {
            Id = int.Parse(parts[0]),
            X = double.Parse(parts[1]),
            Y = double.Parse(parts[2]),
            Z = double.Parse(parts[3])
        });
    }
    return nodeList;
}

private async Task<List<Element>> LoadElementsFromTxtAsync(string filePath)
{
    var elementList = new List<Element>();
    using StreamReader reader = new StreamReader(filePath);
    string line;
    while ((line = await reader.ReadLineAsync()) != null)
    {
        string[] parts = line.Split(' ');
        if (parts.Length < 2) continue;

        if (!Enum.TryParse(parts[0], out ElementType type)) continue;

        var nodeIds = new List<int>();
        for (int i = 1; i < parts.Length; i++)
            nodeIds.Add(int.Parse(parts[i]));

        elementList.Add(new Element
        {
            Type = type,
            NodeIds = nodeIds
        });
    }
    return elementList;
}
