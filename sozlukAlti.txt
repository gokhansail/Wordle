private void GetVTKCellInfo(ElementType type, out int vtkCellType, out int[] nodeOrder)
{
    switch (type)
    {
        case ElementType.tri3:
        case ElementType.tri6:
            vtkCellType = (int)VTKCellType.VTK_TRIANGLE;
            nodeOrder = new[] { 0, 1, 2 };
            break;

        case ElementType.quad4:
        case ElementType.quad8:
            vtkCellType = (int)VTKCellType.VTK_QUAD;
            nodeOrder = new[] { 0, 1, 2, 3 };
            break;

        case ElementType.hexa8:
        case ElementType.hexa20:
        case ElementType.hexa27:
            vtkCellType = (int)VTKCellType.VTK_HEXAHEDRON;
            nodeOrder = new[] { 0, 3, 7, 4, 1, 2, 6, 5 };
            break;

        case ElementType.tetra4:
        case ElementType.tetra10:
            vtkCellType = (int)VTKCellType.VTK_TETRA;
            nodeOrder = new[] { 0, 1, 2, 3 };
            break;

        case ElementType.pyramid5:
        case ElementType.pyramid13:
            vtkCellType = (int)VTKCellType.VTK_PYRAMID;
            nodeOrder = new[] { 0, 1, 2, 3, 4 };
            break;

        case ElementType.prism6:
        case ElementType.prism15:
            vtkCellType = (int)VTKCellType.VTK_WEDGE;
            nodeOrder = new[] { 5, 3, 4, 2, 0, 1 };
            break;

        default:
            throw new NotSupportedException("Desteklenmeyen element tipi: " + type);
    }
}
