private void InitializeToolStrip()
{
    // ToolStripContainer zaten Designer'da üst kısma eklendi varsayılıyor
    ToolStrip toolStrip = new ToolStrip();
    toolStrip.GripStyle = ToolStripGripStyle.Hidden;

    // ▼ Selection Mode (Box / Single)
    ToolStripDropDownButton selectionModeButton = new ToolStripDropDownButton("Seçim Modu");
    ToolStripMenuItem boxItem = new ToolStripMenuItem("Box Selection");
    ToolStripMenuItem singleItem = new ToolStripMenuItem("Single Selection");

    boxItem.Click += (s, e) =>
    {
        selector.Mode = SelectionMode.Box;
        boxItem.Checked = true;
        singleItem.Checked = false;
    };

    singleItem.Click += (s, e) =>
    {
        selector.Mode = SelectionMode.Single;
        singleItem.Checked = true;
        boxItem.Checked = false;
    };

    boxItem.Checked = true; // default olarak box açık
    selectionModeButton.DropDownItems.Add(boxItem);
    selectionModeButton.DropDownItems.Add(singleItem);

    // ▼ Selection Target (Node / Element)
    ToolStripDropDownButton selectionTargetButton = new ToolStripDropDownButton("Seçim Tipi");
    ToolStripMenuItem nodeItem = new ToolStripMenuItem("Node");
    ToolStripMenuItem elementItem = new ToolStripMenuItem("Element");

    nodeItem.Click += (s, e) =>
    {
        selector.Target = SelectionTarget.Node;
        nodeItem.Checked = true;
        elementItem.Checked = false;
    };

    elementItem.Click += (s, e) =>
    {
        selector.Target = SelectionTarget.Element;
        elementItem.Checked = true;
        nodeItem.Checked = false;
    };

    elementItem.Checked = true; // default olarak element açık
    selectionTargetButton.DropDownItems.Add(nodeItem);
    selectionTargetButton.DropDownItems.Add(elementItem);

    // ToolStrip'e butonları ekle
    toolStrip.Items.Add(selectionModeButton);
    toolStrip.Items.Add(selectionTargetButton);

    // ToolStripContainer'a ekle
    toolStripContainer1.TopToolStripPanel.Controls.Add(toolStrip);
}
